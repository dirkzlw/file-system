# file-system
文件管理系统

1.date              --输出当前时间
    date            --Sun Aug 25 09:46:49 CST 2019
    date -f         --2019-08-25 09:46:56

2.pwd               --输出当前绝对路径

3.mkdir             --创建目录
    mkdir filename

4.ll                --输出当前文件列表

5.cd                --切换目录
    cd /
    cd ..
    cd filename

6.rm
    rm -f           --删除文件
    rm -d           --删除目录

7.vi
    vi filename     --编辑文件

8.cp
    cp old new      --复制文件或者目录
